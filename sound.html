    Music Loop

    Code in Action:

    Music clip from: PartnersInRhyme.com


    Example Code:

    require([
    'Sound'
], function (Sound) {
    var that = this;
    this._music = new Sound('audio/musicDemo.wav', {
        loop: true
    });

    $('#music_play').click(function () {
        that._music.play();
    });
    $('#music_pause').click(function () {
        that._music.pause();
    });
    $('#music_stop').click(function () {
        that._music.stop();
    });
});
